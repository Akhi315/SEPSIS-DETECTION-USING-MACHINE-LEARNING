<script>
  // Cached live values from Flask (will be rendered server-side)
  const liveValues = {
    Heart_Rate: Number("{{ sensor_data.Heart_Rate }}"),
    SpO2: Number("{{ sensor_data.SpO2 }}"),
    Body_Temperature: Number("{{ sensor_data.Body_Temperature }}"),
    Respiratory_Rate: Number("{{ sensor_data.Respiratory_Rate }}"),
  };

  // Fill form on click
  document.getElementById('btn-fill').addEventListener('click', () => {
    const set = (id, v) => {
      const el = document.getElementById(id);
      if (!el) return;
      el.value = (Number.isFinite(v) ? v : "");
    };
    set('Heart_Rate', liveValues.Heart_Rate);
    set('SpO2', liveValues.SpO2);
    set('Body_Temperature', liveValues.Body_Temperature);
    set('Respiratory_Rate', liveValues.Respiratory_Rate);
    // Also set mode to live if desired
    const mode = document.getElementById('mode');
    if (mode) mode.value = 'live';
  });
</script>
